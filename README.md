# 프로젝트 소개
팀피셜은 스펙이 아닌 ‘함께 일해본 사람들의 진짜 이야기’로 나를 보여주는
소프트스킬 기반의 팀빌딩 플랫폼입니다.

질문 템플릿을 통해 동료들이 기록한 나의 협업 성향, 일하는 태도,
커뮤니케이션 스타일을 투명하게 제공합니다. 

팀피셜은 더 신뢰할 수 있는 협업, 더 조화로운 팀워크로 이어지는 과정
전반을 설계합니다.

# 폴더 구조
```
src
├── app                 # Next.js App Router (페이지 라우팅)
│   ├── (auth)          # 소셜 로그인 & 인증 페이지
│   ├── (main)          # 로그인 후 메인 기능 영역
│   │   ├── mypage      # 마이페이지
│   │   ├── project     # 프로젝트 상세/목록
│   │   ├── team        # 팀 상세/팀 관리
│   │   ├── applied     # 내가 지원한 팀 리스트
│   │   ├── applicant   # 지원자 관리
│   │   ├── question    # 팀피셜 질문 작성
│   │   ├── recruit     # 팀 모집 등록
│   │   └── teamps ylog # 팀피셜록 페이지
│   └── layout.tsx
│
├── components          # 전역 공통 UI 컴포넌트
│   ├── common          # 버튼, 토스트, 로딩 등 기본 UI
│   ├── login           # 로그인 전용 UI
│   └── modal           # 전역 모달 컴포넌트
│
├── constants           # 상수, dropdown options 등
│
├── contexts            # Modal / Toast 등 글로벌 컨텍스트
│
├── hooks               # 재사용 가능한 custom hooks
│   ├── mutation        # API POST/PUT/DELETE 훅
│   └── queries         # API GET 훅
│
├── libs                # API 레이어, axios wrapper, request 함수
│   └── api             # API 엔드포인트 모듈
│
├── schemas             # Zod 스키마 (form validation)
│
├── store               # Zustand 전역 상태 관리
│
├── styles              # 글로벌 스타일, SCSS 변수
│
├── types               # TypeScript 타입 정의
│
└── utils               # 공통 유틸 함수 모음
```
